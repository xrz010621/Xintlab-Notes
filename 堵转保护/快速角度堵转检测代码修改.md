#### 1 [app_config.h]
L85后新增：

```
/* Fast angle-freeze stall: independent of current, covers all sub_modes.
 * Runs in parallel with the 3 s net-progress stall (STALL_DETECT_CYCLES).
 * Triggers in 500 ms when the shaft makes < 0.1° net progress — fast enough
 * for degraded-dampener slippage and the POSITION_TOLERANCE dead zone,
 * slow enough to avoid false-trips on MT6701 jitter (~0.04°). Threshold is
 * conservatively 2.5× the jitter floor. */

#define FAST_STALL_CYCLES       50U        /* 50 × 10 ms = 500 ms */
#define FAST_STALL_PROGRESS_DEG 0.1f       /* minimum net headway within the window */
```
快速角度冻结的**阈值定义**：`FAST_STALL_CYCLES`冻结累计时长>50个周期(500ms)即冻结
**净进度累计阈值**：`FAST_STALL_PROGRESS_DEG`即角度<0.1°时累计。

##### 4 motor_ctrl_update()内
```c
bool motor_ctrl_update(motor_ctrl_t *ctrl, float now_angle)
{
	/*后置speed mode判定*/
    // if (ctrl->sub_mode == SUB_MODE_SPEED)
    // {
    //     return false;
    // }
    
    /* ── Fast angle-freeze stall ─────────────────────────────────────
     * Independent of motor current; runs for ALL sub-modes (POSITION / TIMED / SPEED).
     * Only suppressed during the startup phase.
     * Covers the degraded-dampener slippage case and the
     * POSITION_TOLERANCE dead zone (0.3°–1.5°) that the net-progress
     * stall explicitly skips. */
    {
	    /* Precompute parameters */
        angle_change = fabsf(now_angle - ctrl->last_angle);
        pos_err = ctrl->target_angle - now_angle;
  
        if (!ctrl->is_startup_phase && (fabsf(pos_err) > POSITION_ARRIVE_DEADBAND))
        {
            if (fabsf(now_angle - ctrl->fast_stall_anchor) >= FAST_STALL_PROGRESS_DEG)
            {
                ctrl->fast_stall_anchor = now_angle;
                ctrl->fast_stall_counter = 0U;
            }
            else
            {
                ctrl->fast_stall_counter++;
            }
        }
        else
        {
            ctrl->fast_stall_anchor = now_angle;
            ctrl->fast_stall_counter = 0U;
        }
  
        if (ctrl->fast_stall_counter > FAST_STALL_CYCLES)
        {
            motor_ctrl_enter_fault(ctrl, now_angle, MOTOR_FAULT_STALL);
            return true;
        }
    }

    /* SPEED mode: fast freeze ran above. New angle update. */

    if (ctrl->sub_mode == SUB_MODE_SPEED)
    {
        ctrl->last_angle = now_angle;
        return false;
    }
}
```
